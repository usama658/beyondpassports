# Tools (#3) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the two client-side tools — a visa checker (reads the shared visa JSON) and a canvas photo maker — each server-rendering a crawlable spec table then hydrating an interactive widget.

**Architecture:** Tool logic is **pure JS** (`assets/js/*-core.js`) exporting functions that take data + inputs and return a result object — tested under Node with a dependency-free assert runner, then loaded in the browser by a thin hydrator (`*-widget.js`) enqueued only on the two tool pages. The checker reuses `data/visas/*.json` (#2). The photo maker uses `data/photo-specs.json`. SEO tables are static HTML in the Elementor page (config-and-verify).

**Tech Stack:** Vanilla JS (ES modules for Node test / global for browser), Node ≥18 (`node --test`-free tiny runner), Canvas API, WordPress/Elementor + hello-child (Foundation #1), shared visa JSON (#2).

**Prerequisite:** #1 Foundation + #2 Content silos done (`data/visas/*.json` exists). `node -v` ≥18 on PATH. Paths relative to repo root.

// Pure checker logic. Works in Node (export) and browser (window.UKVChecker).
export function checkVisa(data, nationality) {
  const v = (data && data.visa) || {};
  if (String(nationality).toUpperCase() !== 'GB') {
    return {
      authoritative: false,
      required: null,
      type: null,
      message: 'Visa rules differ by nationality. Please check the official source for your country.',
      requirements: [],
      ctaHref: null,
    };
  }
  if (!v.required_for_uk) {
    return {
      authoritative: true,
      required: false,
      type: v.type || 'visa-free',
      message: `No visa needed for UK citizens${v.max_stay_days ? ' for up to ' + v.max_stay_days + ' days' : ''}.`,
      requirements: v.requirements || [],
      ctaHref: null,
    };
  }
  return {
    authoritative: true,
    required: true,
    type: v.type || 'visa',
    message: `${v.type || 'Visa'} required for UK citizens${v.max_stay_days ? ' — up to ' + v.max_stay_days + ' days' : ''}.`,
    requirements: v.requirements || [],
    ctaHref: `/apply?dest=${data.slug}&product=visa`,
  };
}

// Browser global (when not in a module context)
if (typeof window !== 'undefined') { window.UKVChecker = { checkVisa }; }
```

// Pure photo geometry. mm -> pixels at a given DPI; output canvas dimensions for a spec.
export function mmToPx(mm, dpi) { return Math.round((mm / 25.4) * dpi); }

export function outputDimensions(spec) {
  const [wmm, hmm] = spec.size_mm;
  return { w: mmToPx(wmm, spec.dpi), h: mmToPx(hmm, spec.dpi) };
}

if (typeof window !== 'undefined') { window.UKVPhoto = { mmToPx, outputDimensions }; }

// Hydrates #ukv-checker. Expects checker-core.js loaded first (window.UKVChecker).
(function () {
  const root = document.getElementById('ukv-checker');
  if (!root || !window.UKVChecker) return;
  const dataDir = root.getAttribute('data-dir'); // e.g. /wp-content/themes/hello-child/data/visas
  const destSel = root.querySelector('[name=dest]');
  const natSel = root.querySelector('[name=nationality]');
  const out = root.querySelector('.ukv-checker-result');

  async function run() {
    const slug = destSel.value;
    out.innerHTML = 'Checking…';
    try {
      const res = await fetch(`${dataDir}/${slug}.json`, { cache: 'force-cache' });
      if (!res.ok) throw new Error('not found');
      const data = await res.json();
      const r = window.UKVChecker.checkVisa(data, natSel.value);
      const reqs = r.requirements.map(x => `<li>${x}</li>`).join('');
      out.innerHTML =
        `<p class="ukv-status">${r.message}</p>` +
        (reqs ? `<ul>${reqs}</ul>` : '') +
        (r.ctaHref ? `<a class="ukv-cta elementor-button" href="${r.ctaHref}">Start your application</a>` : '');
    } catch (e) {
      out.innerHTML = `<p>We don't cover this destination yet. <a href="/contact/">Contact us</a>.</p>`;
    }
  }
  destSel.addEventListener('change', run);
  natSel.addEventListener('change', run);
  root.querySelector('.ukv-checker-go')?.addEventListener('click', run);
})();
```

```js
// Hydrates #ukv-photo. Client-side only — no upload. Needs photo-core.js (window.UKVPhoto).
(function () {
  const root = document.getElementById('ukv-photo');
  if (!root || !window.UKVPhoto) return;
  const fileIn = root.querySelector('[name=file]');
  const presetSel = root.querySelector('[name=preset]');
  const zoom = root.querySelector('[name=zoom]');
  const canvas = root.querySelector('canvas');
  const dl = root.querySelector('.ukv-photo-download');
  const specs = JSON.parse(root.getAttribute('data-specs')); // injected by PHP
  let img = null;

  function draw() {
    if (!img) return;
    const spec = specs[presetSel.value];
    const { w, h } = window.UKVPhoto.outputDimensions(spec);
    canvas.width = w; canvas.height = h;
    const ctx = canvas.getContext('2d');
    ctx.fillStyle = spec.bg === 'white' ? '#ffffff' : '#f0f0f0';
    ctx.fillRect(0, 0, w, h);
    const scale = (parseFloat(zoom.value) || 1) * Math.max(w / img.width, h / img.height);
    const dw = img.width * scale, dh = img.height * scale;
    ctx.drawImage(img, (w - dw) / 2, (h - dh) / 2, dw, dh);
  }
  function refreshDownload() {
    canvas.toBlob(b => { dl.href = URL.createObjectURL(b); dl.download = `visa-photo-${presetSel.value}.jpg`; }, 'image/jpeg', 0.95);
  }
  fileIn.addEventListener('change', e => {
    const f = e.target.files[0]; if (!f) return;
    const fr = new FileReader();
    fr.onload = () => { img = new Image(); img.onload = () => { draw(); refreshDownload(); }; img.src = fr.result; };
    fr.readAsDataURL(f); // stays in-browser
  });
  [presetSel, zoom].forEach(el => el.addEventListener('input', () => { draw(); refreshDownload(); }));
})();
```

`wordpress/hello-child/inc/tools-enqueue.php`:
```php
<?php
// Load tool JS only on the two tool pages (keeps CWV budget from Foundation).
add_action('wp_enqueue_scripts', function () {
    $dir = get_stylesheet_directory_uri() . '/assets/js';
    if (is_page('do-i-need-a-visa')) {
        wp_enqueue_script('ukv-checker-core', "$dir/checker-core.js", [], '1.0', true);
        wp_enqueue_script('ukv-checker-widget', "$dir/checker-widget.js", ['ukv-checker-core'], '1.0', true);
    }
    if (is_page('visa-photo')) {
        wp_enqueue_script('ukv-photo-core', "$dir/photo-core.js", [], '1.0', true);
        wp_enqueue_script('ukv-photo-widget', "$dir/photo-widget.js", ['ukv-photo-core'], '1.0', true);
    }
});
```
> Note: `checker-core.js` uses an ES `export`; for classic enqueue, the browser global block at the file end is what the widget consumes. Load it as a normal script (the `export` line is ignored by non-module script tags only if stripped). To avoid parser errors, add `add_filter('script_loader_tag', ...)` to mark these as `type="module"`, OR keep a browserstrip: simplest is to serve the core as a module. Implement the module tag filter:
```php
add_filter('script_loader_tag', function ($tag, $handle) {
    if (in_array($handle, ['ukv-checker-core', 'ukv-photo-core'], true)) {
        $tag = str_replace('<script ', '<script type="module" ', $tag);
    }
    return $tag;
}, 10, 2);
```
Put both blocks in the same file.
